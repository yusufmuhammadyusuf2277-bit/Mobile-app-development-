Lesson 26: Form Validation


TextFormField(
  decoration: InputDecoration(labelText: 'Email'),
  validator: (value) {
    if (value!.isEmpty) return 'Email required';
    if (!value.contains('@')) return 'Invalid email';
    return null;
  },
)
