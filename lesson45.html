Lesson 45: Custom Widgets


class MyCard extends StatelessWidget {
  final String title;

  MyCard({required this.title});

  @override
  Widget build(BuildContext context) {
    return Card(
      child: Padding(
        padding: EdgeInsets.all(16),
        child: Text(title),
      ),
    );
  }
}
